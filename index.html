function doPost(e){
  const ss=SpreadsheetApp.openById('SHEET_ID_PLACEHOLDER');
  const data=JSON.parse(e.postData.contents);
  const cfg={
    students:{name:'Students',headers:['ID','Name','Phone','Course','Fees','Start Date','Status']},
    attendance:{name:'Attendance',headers:['ID','Date','Student','Time Slot','Status']},
    payments:{name:'Payments',headers:['ID','Date','Student','Amount','Method','Note']},
    enquiries:{name:'Enquiries',headers:['Date','Name','Phone','Course','Status','Notes']},
    rto:{name:'RTO Services',headers:['Student','Type','Applied Date','Test Date','Status']},
    rtowalk:{name:'Other RTO',headers:['ID','Date','Name','Mobile','Service','Total Fee','Paid','Method','Status','Pending','Notes']},
    fleet:{name:'Fleet Log',headers:['ID','Date','Car','Instructor','Hours','Fuel(L)','Fuel Cost','Odometer','Issue']}
  };
  for(const key in cfg){
    let sheet=ss.getSheetByName(cfg[key].name);
    if(!sheet){sheet=ss.insertSheet(cfg[key].name);}
    sheet.clearContents();
    sheet.getRange(1,1,1,cfg[key].headers.length).setValues([cfg[key].headers]).setFontWeight('bold').setBackground('#ff7c01').setFontColor('#ffffff');
    if(data[key]&&data[key].length>0)
      sheet.getRange(2,1,data[key].length,cfg[key].headers.length).setValues(data[key]);
  }
  return ContentService.createTextOutput(JSON.stringify({status:'success'})).setMimeType(ContentService.MimeType.JSON);
}
